
//patientSqlDAO

CREATE TABLE patients (
	id INT PRIMARY KEY AUTO_INCREMENT,
    FirstName VARCHAR(255) NOT NULL,
    LastName VARCHAR(255) NOT NULL,
    ID_NUMBER VARCHAR(255) UNIQUE,
    Password VARCHAR(255),
    City VARCHAR(255),
    Birth_Date VARCHAR(255),
    Sex VARCHAR(255),
    Address VARCHAR(255),
    Mobile_Number VARCHAR(255),
    Diseases VARCHAR(255),
    Alergies VARCHAR(255)
);

//doctorSqlDAO
CREATE TABLE doctors (
	id INT PRIMARY KEY AUTO_INCREMENT,
    FirstName VARCHAR(255) NOT NULL,
    LastName VARCHAR(255) NOT NULL,
    ID_NUMBER VARCHAR(255) UNIQUE,
    Password VARCHAR(255),
    City VARCHAR(255),
    Doctor_Specialities VARCHAR(255),
    Mobile_Number VARCHAR(255),
    Doctor_Qualifications VARCHAR(255),
    Experience VARCHAR(255),
    Graduation VARCHAR(255)
    );

CREATE TABLE history (
	id INT PRIMARY KEY AUTO_INCREMENT,
    Patient_Name VARCHAR(255) NOT NULL,
	Patient_LastName VARCHAR(255) NOT NULL,
    Patient_ID_NUMBER VARCHAR(255) NOT NULL,
    Doctor_Name VARCHAR(255) NOT NULL,
    Doctor_LastName VARCHAR(255) NOT NULL,
    Doctor_ID_NUMBER VARCHAR(255) NOT NULL,
    date VARCHAR(255),
    Reason VARCHAR(255),
    Type VARCHAR(255),
    Medical_Conclusion VARCHAR(255)
);
